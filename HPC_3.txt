#include <iostream>
#include <omp.h>

using namespace std;

int main() {
    int n;

    cout << "Enter size: ";
    cin >> n;

    int arr[n];

    cout << "Enter elements:\n";
    for (int i = 0; i < n; i++)
        cin >> arr[i];

    int min_val = arr[0];
    int max_val = arr[0];
    int sum = 0;
    float avg;

    #pragma omp parallel for reduction(min:min_val)
    for (int i = 0; i < n; i++) {
        if (arr[i] < min_val)
            min_val = arr[i];
    }

    #pragma omp parallel for reduction(max:max_val)
    for (int i = 0; i < n; i++) {
        if (arr[i] > max_val)
            max_val = arr[i];
    }

    #pragma omp parallel for reduction(+:sum)
    for (int i = 0; i < n; i++) {
        sum += arr[i];
    }

    avg = (float)sum / n;

    cout << "\nMinimum = " << min_val;
    cout << "\nMaximum = " << max_val;
    cout << "\nSum = " << sum;
    cout << "\nAverage = " << avg;

    return 0;
}
