#include <iostream>
#include <vector>
#include <queue>
#include <stack>
#include <omp.h>

using namespace std;

class Graph {
    int V;
    vector<vector<int>> adj;

public:
    Graph(int v) {
        V = v;
        adj.resize(V);
    }

    void addEdge(int u, int v) {
        adj[u].push_back(v);
        adj[v].push_back(u);
    }

    void BFS(int start) {
        vector<bool> vis(V, false);
        queue<int> q;

        q.push(start);
        vis[start] = true;

        while (!q.empty()) {
            int node = q.front();
            q.pop();

            cout << node << " ";

            #pragma omp parallel for
            for (int i = 0; i < adj[node].size(); i++) {
                int nbr = adj[node][i];

                #pragma omp critical
                {
                    if (!vis[nbr]) {
                        vis[nbr] = true;
                        q.push(nbr);
                    }
                }
            }
        }
    }

    void DFS(int start) {
        vector<bool> vis(V, false);
        stack<int> st;

        st.push(start);

        while (!st.empty()) {
            int node = st.top();
            st.pop();

            if (vis[node]) continue;

            vis[node] = true;
            cout << node << " ";

            #pragma omp parallel for
            for (int i = 0; i < adj[node].size(); i++) {
                int nbr = adj[node][i];

                #pragma omp critical
                {
                    if (!vis[nbr])
                        st.push(nbr);
                }
            }
        }
    }
};

int main() {
    int V, E, u, v, start;

    cout << "Enter vertices and edges: ";
    cin >> V >> E;

    Graph g(V);

    cout << "Enter edges:\n";
    for (int i = 0; i < E; i++) {
        cin >> u >> v;
        g.addEdge(u, v);
    }

    cout << "Enter start vertex: ";
    cin >> start;

    cout << "BFS: ";
    g.BFS(start);

    cout << "\nDFS: ";
    g.DFS(start);

    return 0;
}
