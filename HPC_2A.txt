#include <iostream>
#include <ctime>
#include <omp.h>

using namespace std;

void bubbleSort(int a[], int n) {
    for (int i = 0; i < n - 1; i++) {

        #pragma omp parallel for
        for (int j = 0; j < n - i - 1; j++) {
            if (a[j] > a[j + 1])
                swap(a[j], a[j + 1]);
        }
    }
}

int main() {
    int n;

    cout << "Enter size: ";
    cin >> n;

    int a[n];

    cout << "Enter elements:\n";
    for (int i = 0; i < n; i++)
        cin >> a[i];

    clock_t start = clock();

    bubbleSort(a, n);

    clock_t end = clock();

    cout << "Sorted Array:\n";
    for (int i = 0; i < n; i++)
        cout << a[i] << " ";

    double time_taken = double(end - start) / CLOCKS_PER_SEC;

    cout << "\nTime: " << time_taken << " sec";

    return 0;
}
